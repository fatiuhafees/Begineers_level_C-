// Format Specifier % = defines and formats a type of data to be displayed 

//<stdbool.h>, used for working with bool
//<ctype.h>,   used for working with string (string and char)

//%c = char 
//%s = string (array of characters)
//%f = float
//%lf = double 
//%d = integer 
//%d = bool

//%.1 = decimal precision 
//%1 = minium field width 
//%- = left allign 

#include <stdio.h>
float item1 = 2.3222;
float item2 = 12.3222;

int main(){
//decimal precision %.1
     printf("Item 1: %.2f",item1);  //ouput: Item 1: 2.32

//minium field width %1
     printf("Item 1: %8.2f/n",item1);  //ouput: Item 1:         2.32
     printf("Item 1: %8.2f/n",item2);  //ouput: Item 2:        12.32

//left align %-1
     printf("Item 1: %-8.2f",item1);  //output: Item 1: 2.32
     printf("Item 2: %-8.2f/n",item2);//ouput: Item 2: 12.32
     return 0;
}

#include <stdio.h>

{ 
   char array[] = {'F', 'A', 'D', 'B', 'C'};
   int size = sizeof(array)/sizeof(array[0]);

   sort(array, size);
   printArray(array, size);
 
   return 0; 
}

void sort(char array[], int size)
{
//we do 'size -1' as we want an index number, not an integer number. idex = 4, int = 5. 
   for(int i = 0; i < size - 1; i++)
   {
      for(int j = 0; j < (size -i )- 1; j++)
      {
//if the element to the right is bigger do...
         if(array[j] > array[j+1]) 
         {
            int temp = array[j];
            array[j] = array[j+1];
            array[j+1] = temp;
         }
      }
   }
}

void printArray(char array[], int size)
{
   for(int i = 0; i < size; i++)
   {
      printf("%c ", array[i]);
   }
}

// Bitwise Operators = special operators used in bit-level programming (knowing binary is important for this topic)

// & = AND. Both decimals have to have 1. 
// | = OR.  Either decimal can have 1.
// ^ = XOR. Only one decimal should have 1.
// <<  left shift. Copies a decimal and shifts to the left n amount of times. 
// >>  right shift. Copies a decimal and shifts to the right n amount of times.

#include <stdio.h>
 
int main()
{
   int x = 6;  //    6  = 00000110
   int y = 12; //    12 = 00001100 
   int z = 0;  //    0  = 00000000

   z = x & y;
   printf("AND = %d\n", z); //output: 4

   z = x | y;
   printf("OR = %d\n", z); //output: 14

   z = x ^ y;
   printf("XOR = %d\n", z); //output: 10

   z = x << 2;
   printf("SHIFT LEFT = %d\n", z); //output: 10

   z = x >> 2;
   printf("SHIFT RIGHT = %d\n", z);

   return 0;
}




//Memory Address

// memory block = a single unit (byte) within memory (house), used to hold some value (person)
// memory address = the address of where a memory block is located (house address)

//memory block = printf("%d", sizeof(...))
//memory address = printf("%p", &...);

#include <stdio.h>

int main(){
   int num = 4; 
   char a;
   char b[4];

//Pointers
// pointer = a "variable-like" reference that holds a memory address to another variable, array, etc.
// Some tasks are performed more easily with pointers
// pointers have the same memory address and value as the variable it holds
// pointers have a different memory block as the variable it holds with a byte of 8

// * = indirection operator (value at address)
// pointer: datatype *p... = &...;

// Dereferencing refers to assessing data/value contained in memory pointed to by a pointer

#include <stdio.h>

int main()
{

   int age = 21;
//naming convention for pointers: "*p" with first letter upper case
   int *pAge = &age;

//same memory address
   printf("address of age: %p\n", &age);
   printf("value of pAge: %p\n", &pAge);

//different bytes
   printf("size of age: %d bytes\n", sizeof(age));
   printf("size of pAge: %d bytes\n", sizeof(pAge));

//same value 
   printf("value of age: %d\n", age);
   printf("value at stored address: %d\n", *pAge); //dereferencing

//pointers can be used in functions like a normal variable would
   printAge(pAge); 
   return 0;
}

void printAge(int *pAge)
{
   printf("Age: %d\n", *pAge); //dereference
}
